Selecting elements and operators 
d3.select("selector");
d3.selectAll("selector");
							Set attribute of element:
							d3.select("selector").attr("class", "green")
						
							Set style of element:
							d3.select("selector").style("width", "120px")
						
							Set content of element: 				
							d3.select("selector").text("Text example")
						
							Set HTML-code of element:							
							d3.select("selector").html("
Title
")
HTML Elements control 
					Add element:							
					d3.select("selector").append("item").text("Title");
				
					Remove element:							
					d3.select("selector").remove("item");
Data 
					
					d3.blob("URL") //Get a file as a blob
				
				
					d3.buffer("URL") //Get a file as an array buffer
				
							
					d3.csv("URL") //Get a comma-separated values (CSV) file:
				
					
					d3.dsv("URL") //Get a delimiter-separated values (CSV) file
				
			
					d3.html("URL") //Get an HTML file
				
			
					d3.image("URL") //Get an image
				
			
					d3.json("URL") //Get a JSON file
				
			
					d3.text("URL") //Get a plain text file
				
			
					d3.xml("URL") //Get  an XML file.
